58 void qualrout_init()
59 //
60 // Input: none
61 // Output: none
62 // Purpose: initializes water quality concentrations in all nodes and links.
63 //
64 {
65  int i, p, isWet;
66  double c;
67 
68  for (i = 0; i < Nobjects[NODE]; i++)
69  {
70  isWet = ( Node[i].newDepth > FUDGE );
71  for (p = 0; p < Nobjects[POLLUT]; p++)
72  {
73  c = 0.0;
74  if ( isWet ) c = Pollut[p].initConcen;
75  Node[i].oldQual[p] = c;
76  Node[i].newQual[p] = c;
77  }
78  }
79 
80  for (i = 0; i < Nobjects[LINK]; i++)
81  {
82  isWet = ( Link[i].newDepth > FUDGE );
83  for (p = 0; p < Nobjects[POLLUT]; p++)
84  {
85  c = 0.0;
86  if ( isWet ) c = Pollut[p].initConcen;
87  Link[i].oldQual[p] = c;
88  Link[i].newQual[p] = c;
89  }
90  }
91 }

95 void qualrout_execute(double tStep)
96 //
97 // Input: tStep = routing time step (sec)
98 // Output: none
99 // Purpose: routes water quality constituents through the drainage
100 // network over the current time step.
101 //
102 {
103  int i, j;
104  double qIn, vAvg;
105 
106  // --- find mass flow each link contributes to its downstream node
107  for ( i = 0; i < Nobjects[LINK]; i++ ) findLinkMassFlow(i, tStep);
108 
109  // --- find new water quality concentration at each node
110  for (j = 0; j < Nobjects[NODE]; j++)
111  {
112  // --- get node inflow and average volume
113  qIn = Node[j].inflow;
114  vAvg = (Node[j].oldVolume + Node[j].newVolume) / 2.0;
115 
116  // --- save inflow concentrations if treatment applied
117  if ( Node[j].treatment )
118  {
119  if ( qIn < ZERO ) qIn = 0.0;
120  treatmnt_setInflow(qIn, Node[j].newQual);
121  }
122 
123  // --- find new quality at the node
124  if ( Node[j].type == STORAGE || Node[j].oldVolume > FUDGE )
125  {
126  findStorageQual(j, tStep);
127  }
128  else findNodeQual(j);
129 
130  // --- apply treatment to new quality values
131  if ( Node[j].treatment ) treatmnt_treat(j, qIn, vAvg, tStep);
132  }
133 
134  // --- find new water quality in each link
135  for ( i = 0; i < Nobjects[LINK]; i++ ) findLinkQual(i, tStep);
136 }

140 double getMixedQual(double c, double v1, double wIn, double qIn, double tStep)
141 //
142 // Input: c = concentration in reactor at start of time step (mass/ft3)
143 // v1 = volume in reactor at start of time step (ft3)
144 // wIn = mass inflow rate (mass/sec)
145 // qIn = flow inflow rate (cfs)
146 // tStep = time step (sec)
147 // Output: returns pollutant concentration at end of time step (mass/ft3)
148 // Purpose: finds pollutant concentration within a completely mixed reactor.
149 //
150 {
151  double vIn, cIn, cMax;
152 
153  // --- if no inflow then reactor concentration is unchanged
154  if ( qIn <= ZERO ) return c;
155 
156  // --- compute concentration of any inflow
157  vIn = qIn * tStep;
158  cIn = wIn * tStep / vIn;
159 
160  // --- mixture concen. can't exceed either original or inflow concen.
161  cMax = MAX(c, cIn);
162 
163  // --- mix inflow with current reactor contents
164  c = (c*v1 + wIn*tStep) / (v1 + vIn);
165  c = MIN(c, cMax);
166  c = MAX(c, 0.0);
167  return c;
168 }

173 void findLinkMassFlow(int i, double tStep)
174 //
175 // Input: i = link index
176 // tStep = time step (sec)
177 // Output: none
178 // Purpose: adds constituent mass flow out of link to the total
179 // accumulation at the link's downstream node.
180 //
181 // Note: Node[].newQual[], the accumulator variable, already contains
182 // contributions from runoff and other external inflows from
183 // calculations made in routing_execute().
184 {
185  int j, p;
186  double qLink, w;
187 
188  // --- find inflow to downstream node
189  qLink = Link[i].newFlow;
190 
191  // --- identify index of downstream node
192  j = Link[i].node2;
193  if ( qLink < 0.0 ) j = Link[i].node1;
194  qLink = fabs(qLink);
195 
196  // --- examine each pollutant
197  for (p = 0; p < Nobjects[POLLUT]; p++)
198  {
199  // --- temporarily accumulate inflow load in Node[j].newQual
200  w = qLink * Link[i].oldQual[p];
201  Node[j].newQual[p] += w;
202 
203  // --- update total load transported by link
204  Link[i].totalLoad[p] += w * tStep;
205  }
206 }

210 void findNodeQual(int j)
211 //
212 // Input: j = node index
213 // Output: none
214 // Purpose: finds new quality in a node with no storage volume.
215 //
216 {
217  int p;
218  double qNode;
219 
220  // --- if there is flow into node then concen. = mass inflow/node flow
221  qNode = Node[j].inflow;
222  if ( qNode > ZERO )
223  {
224  for (p = 0; p < Nobjects[POLLUT]; p++)
225  {
226  Node[j].newQual[p] /= qNode;
227  }
228  }
229 
230  // --- otherwise concen. is 0
231  else for (p = 0; p < Nobjects[POLLUT]; p++) Node[j].newQual[p] = 0.0;
232 }

236 void findLinkQual(int i, double tStep)
237 //
238 // Input: i = link index
239 // tStep = routing time step (sec)
240 // Output: none
241 // Purpose: finds new quality in a link at end of the current time step.
242 //
243 {
244  int j, // upstream node index
245  k, // conduit index
246  p; // pollutant index
247  double wIn, // pollutant mass inflow rate (mass/sec)
248  qIn, // inflow rate (cfs)
249  qSeep, // rate of seepage loss (cfs)
250  v1, // link volume at start of time step (ft3)
251  v2, // link volume at end of time step (ft3)
252  c1, // current concentration within link (mass/ft3)
253  c2, // new concentration within link (mass/ft3)
254  vEvap, // volume lost to evaporation (ft3)
255  vLosses, // evap. + seepage volume loss (ft3)
256  fEvap, // evaporation concentration factor
257  barrels; // number of barrels in conduit
258 
259  // --- identify index of upstream node
260  j = Link[i].node1;
261  if ( Link[i].newFlow < 0.0 ) j = Link[i].node2;
262 
263  // --- link quality is that of upstream node when
264  // link is not a conduit or is a dummy link
265  if ( Link[i].type != CONDUIT || Link[i].xsect.type == DUMMY )
266  {
267  for (p = 0; p < Nobjects[POLLUT]; p++)
268  {
269  Link[i].newQual[p] = Node[j].newQual[p];
270  }
271  return;
272  }
273 
274  // --- get flow rates and evaporation loss
275  k = Link[i].subIndex;
276  barrels = Conduit[k].barrels;
277  qIn = fabs(Conduit[k].q1) * barrels;
278  qSeep = Conduit[k].seepLossRate * barrels;
279  vEvap = Conduit[k].evapLossRate * barrels * tStep;
280 
281  // --- get starting and ending volumes
282  v1 = Link[i].oldVolume;
283  v2 = Link[i].newVolume;
284  vLosses = qSeep*tStep + vEvap;
285 
286  // --- compute factor by which concentrations are increased due to
287  // evaporation loss
288  fEvap = 1.0;
289  if ( vEvap > 0.0 && v1 > ZeroVolume ) fEvap += vEvap / v1;
290 
291  // --- Steady Flow routing requires special treatment
292  if ( RouteModel == SF )
293  {
294  findSFLinkQual(i, qSeep, fEvap, tStep);
295  return;
296  }
297 
298  // --- adjust inflow to compensate for volume change under Dynamic
299  // Wave routing (which produces just a single (out)flow rate
300  // for a conduit)
301  if ( RouteModel == DW )
302  {
303  qIn = qIn + (v2 + vLosses - v1) / tStep;
304  qIn = MAX(qIn, 0.0);
305  }
306 
307  // --- examine each pollutant
308  for (p = 0; p < Nobjects[POLLUT]; p++)
309  {
310  // --- start with concen. at start of time step
311  c1 = Link[i].oldQual[p];
312 
313  // --- update mass balance accounting for seepage loss
314  massbal_addSeepageLoss(p, qSeep*c1);
315 
316  // --- increase concen. by evaporation factor
317  c1 *= fEvap;
318 
319  // --- reduce concen. by 1st-order reaction
320  c2 = getReactedQual(p, c1, v1, tStep);
321 
322  // --- mix resulting contents with inflow from upstream node
323  wIn = Node[j].newQual[p]*qIn;
324  c2 = getMixedQual(c2, v1, wIn, qIn, tStep);
325 
326  // --- set concen. to zero if remaining volume is negligible
327  if ( v2 < ZeroVolume )
328  {
329  massbal_addToFinalStorage(p, c2 * v2);
330  c2 = 0.0;
331  }
332 
333  // --- assign new concen. to link
334  Link[i].newQual[p] = c2;
335  }
336 }

340 void findSFLinkQual(int i, double qSeep, double fEvap, double tStep)
341 //
342 // Input: i = link index
343 // tStep = routing time step (sec)
344 // Output: none
345 // Purpose: finds new quality in a link at end of the current time step for
346 // Steady Flow routing.
347 //
348 {
349  int j = Link[i].node1;
350  int p;
351  double c1, c2;
352  double lossRate;
353 
354  // --- examine each pollutant
355  for (p = 0; p < Nobjects[POLLUT]; p++)
356  {
357  // --- conduit's quality equals upstream node quality
358  c1 = Node[j].newQual[p];
359 
360  // --- update mass balance accounting for seepage loss
361  massbal_addSeepageLoss(p, qSeep*c1);
362 
363  // --- increase concen. by evaporation factor
364  c1 *= fEvap;
365 
366  // --- apply first-order decay over travel time
367  c2 = c1;
368  if ( Pollut[p].kDecay > 0.0 )
369  {
370  c2 = c1 * exp(-Pollut[p].kDecay * tStep);
371  c2 = MAX(0.0, c2);
372  lossRate = (c1 - c2) * Link[i].newFlow;
373  massbal_addReactedMass(p, lossRate);
374  }
375  Link[i].newQual[p] = c2;
376  }
377 }

381 void findStorageQual(int j, double tStep)
382 //
383 // Input: j = node index
384 // tStep = routing time step (sec)
385 // Output: none
386 // Purpose: finds new quality in a node with storage volume.
387 //
388 {
389  int p, // pollutant index
390  k; // storage unit index
391  double qIn, // inflow rate (cfs)
392  wIn, // pollutant mass inflow rate (mass)
393  v1, // volume at start of time step (ft3)
394  c1, // initial pollutant concentration (mass/ft3)
395  c2, // final pollutant concentration (mass/ft3)
396  qExfil = 0.0, // exfiltration rate from storage unit (cfs)
397  vEvap = 0.0, // evaporation loss from storage unit (ft3)
398  fEvap = 1.0; // evaporation concentration factor
399 
400  // --- get inflow rate & initial volume
401  qIn = Node[j].inflow;
402  v1 = Node[j].oldVolume;
403 
404  // -- for storage nodes
405  if ( Node[j].type == STORAGE )
406  {
407  // --- update hydraulic residence time
408  // (HRT can be used in treatment functions)
409  updateHRT(j, Node[j].oldVolume, qIn, tStep);
410 
411  // --- get exfiltration rate and evaporation loss
412  k = Node[j].subIndex;
413  qExfil = Storage[k].exfilLoss / tStep;
414  vEvap = Storage[k].evapLoss;
415 
416  // --- compute factor by which concentrations are increased due to
417  // evaporation loss (avoiding huge factors as storage unit
418  // dries out completely)
419  if ( vEvap > 0.0 && v1 > ZeroVolume ) fEvap += vEvap / v1;
420  }
421 
422  // --- for each pollutant
423  for (p = 0; p < Nobjects[POLLUT]; p++)
424  {
425  // --- start with concen. at start of time step
426  c1 = Node[j].oldQual[p];
427 
428  // --- update mass balance accounting for exfiltration loss
429  massbal_addSeepageLoss(p, qExfil*c1);
430 
431  // --- increase concen. by evaporation factor
432  c1 *= fEvap;
433 
434  // --- apply first order reaction only if no separate treatment function
435  if ( Node[j].treatment == NULL ||
436  Node[j].treatment[p].equation == NULL )
437  {
438  c1 = getReactedQual(p, c1, v1, tStep);
439  }
440 
441  // --- mix resulting contents with inflow from all sources
442  // (temporarily accumulated in Node[j].newQual)
443  wIn = Node[j].newQual[p];
444  c2 = getMixedQual(c1, v1, wIn, qIn, tStep);
445 
446  // --- set concen. to zero if remaining volume is negligible
447  if ( Node[j].newVolume <= ZeroVolume )
448  {
449  massbal_addToFinalStorage(p, c2 * Node[j].newVolume);
450  c2 = 0.0;
451  }
452 
453  // --- assign new concen. to node
454  Node[j].newQual[p] = c2;
455  }
456 }

460 void updateHRT(int j, double v, double q, double tStep)
461 //
462 // Input: j = node index
463 // v = storage volume (ft3)
464 // q = inflow rate (cfs)
465 // tStep = time step (sec)
466 // Output: none
467 // Purpose: updates hydraulic residence time (i.e., water age) at a
468 // storage node.
469 //
470 {
471  int k = Node[j].subIndex;
472  double hrt = Storage[k].hrt;
473  if ( v < ZERO ) hrt = 0.0;
474  else hrt = (hrt + tStep) * v / (v + q*tStep);
475  Storage[k].hrt = MAX(hrt, 0.0);
476 }

480 double getReactedQual(int p, double c, double v1, double tStep)
481 //
482 // Input: p = pollutant index
483 // c = initial concentration (mass/ft3)
484 // v1 = initial volume (ft3)
485 // tStep = time step (sec)
486 // Output: none
487 // Purpose: applies a first order reaction to a pollutant over a given
488 // time step.
489 //
490 {
491  double c2, lossRate;
492  double kDecay = Pollut[p].kDecay;
493 
494  if ( kDecay == 0.0 ) return c;
495  c2 = c * (1.0 - kDecay * tStep);
496  c2 = MAX(0.0, c2);
497  lossRate = (c - c2) * v1 / tStep;
498  massbal_addReactedMass(p, lossRate);
499  return c2;
500 }
