43 void dwflow_findConduitFlow(int j, int steps, double omega, double dt)
44 //
45 // Input: j = link index
46 // steps = number of iteration steps taken
47 // omega = under-relaxation parameter
48 // dt = time step (sec)
49 // Output: returns new flow value (cfs)
50 // Purpose: updates flow in conduit link by solving finite difference
51 // form of continuity and momentum equations.
52 //
53 {
54  int k; // index of conduit
55  int n1, n2; // indexes of end nodes
56  double z1, z2; // upstream/downstream invert elev. (ft)
57  double h1, h2; // upstream/dounstream flow heads (ft)
58  double y1, y2; // upstream/downstream flow depths (ft)
59  double a1, a2; // upstream/downstream flow areas (ft2)
60  double r1; // upstream hyd. radius (ft)
61  double yMid, rMid, aMid; // mid-stream or avg. values of y, r, & a
62  double aWtd, rWtd; // upstream weighted area & hyd. radius
63  double qLast; // flow from previous iteration (cfs)
64  double qOld; // flow from previous time step (cfs)
65  double aOld; // area from previous time step (ft2)
66  double v; // velocity (ft/sec)
67  double rho; // upstream weighting factor
68  double sigma; // inertial damping factor
69  double length; // effective conduit length (ft)
70  double dq1, dq2, dq3, dq4, dq5, // terms in momentum eqn.
71  dq6; // term for evap and infil losses
72  double denom; // denominator of flow update formula
73  double q; // new flow value (cfs)
74  double barrels; // number of barrels in conduit
75  TXsect* xsect = &Link[j].xsect; // ptr. to conduit's cross section data
76  char isFull = FALSE; // TRUE if conduit flowing full
77  char isClosed = FALSE; // TRUE if conduit closed
78 
79  // --- adjust isClosed status by any control action
80  if ( Link[j].setting == 0 ) isClosed = TRUE;
81 
82  // --- get flow from last time step & previous iteration
83  k = Link[j].subIndex;
84  barrels = Conduit[k].barrels;
85  qOld = Link[j].oldFlow / barrels;
86  qLast = Conduit[k].q1;
87 
88  // --- get most current heads at upstream and downstream ends of conduit
89  n1 = Link[j].node1;
90  n2 = Link[j].node2;
91  z1 = Node[n1].invertElev + Link[j].offset1;
92  z2 = Node[n2].invertElev + Link[j].offset2;
93  h1 = Node[n1].newDepth + Node[n1].invertElev;
94  h2 = Node[n2].newDepth + Node[n2].invertElev;
95  h1 = MAX(h1, z1);
96  h2 = MAX(h2, z2);
97 
98  // --- get unadjusted upstream and downstream flow depths in conduit
99  // (flow depth = head in conduit - elev. of conduit invert)
100  y1 = h1 - z1;
101  y2 = h2 - z2;
102  y1 = MAX(y1, FUDGE);
103  y2 = MAX(y2, FUDGE);
104 
105  // --- flow depths can't exceed full depth of conduit
106  y1 = MIN(y1, xsect->yFull);
107  y2 = MIN(y2, xsect->yFull);
108 
109  // -- get area from solution at previous time step
110  aOld = Conduit[k].a2;
111  aOld = MAX(aOld, FUDGE);
112 
113  // --- use Courant-modified length instead of conduit's actual length
114  length = Conduit[k].modLength;
115 
116  // --- find surface area contributions to upstream and downstream nodes
117  // based on previous iteration's flow estimate
118  findSurfArea(j, qLast, length, &h1, &h2, &y1, &y2);
119 
120  // --- compute area at each end of conduit & hyd. radius at upstream end
121  a1 = getArea(xsect, y1);
122  a2 = getArea(xsect, y2);
123  r1 = getHydRad(xsect, y1);
124 
125  // --- compute area & hyd. radius at midpoint
126  yMid = 0.5 * (y1 + y2);
127  aMid = getArea(xsect, yMid);
128  rMid = getHydRad(xsect, yMid);
129 
130  // --- alternate approach not currently used, but might produce better
131  // Bernoulli energy balance for steady flows
132  //aMid = (a1+a2)/2.0;
133  //rMid = (r1+getHydRad(xsect,y2))/2.0;
134 
135  // --- check if conduit is flowing full
136  if ( y1 >= xsect->yFull &&
137  y2 >= xsect->yFull) isFull = TRUE;
138 
139  // --- set new flow to zero if conduit is dry or if flap gate is closed
140  if ( Link[j].flowClass == DRY ||
141  Link[j].flowClass == UP_DRY ||
142  Link[j].flowClass == DN_DRY ||
143  isClosed ||
144  aMid <= FUDGE )
145  {
146  Conduit[k].a1 = 0.5 * (a1 + a2);
147  Conduit[k].q1 = 0.0;;
148  Conduit[k].q2 = 0.0;
149  Link[j].dqdh = GRAVITY * dt * aMid / length * barrels;
150  Link[j].froude = 0.0;
151  Link[j].newDepth = MIN(yMid, Link[j].xsect.yFull);
152  Link[j].newVolume = Conduit[k].a1 * link_getLength(j) * barrels;
153  Link[j].newFlow = 0.0;
154  return;
155  }
156 
157  // --- compute velocity from last flow estimate
158  v = qLast / aMid;
159  if ( fabs(v) > MAXVELOCITY ) v = MAXVELOCITY * SGN(qLast);
160 
161  // --- compute Froude No.
162  Link[j].froude = link_getFroude(j, v, yMid);
163  if ( Link[j].flowClass == SUBCRITICAL &&
164  Link[j].froude > 1.0 ) Link[j].flowClass = SUPCRITICAL;
165 
166  // --- find inertial damping factor (sigma)
167  if ( Link[j].froude <= 0.5 ) sigma = 1.0;
168  else if ( Link[j].froude >= 1.0 ) sigma = 0.0;
169  else sigma = 2.0 * (1.0 - Link[j].froude);
170 
171  // --- get upstream-weighted area & hyd. radius based on damping factor
172  // (modified version of R. Dickinson's slope weighting)
173  rho = 1.0;
174  if ( !isFull && qLast > 0.0 && h1 >= h2 ) rho = sigma;
175  aWtd = a1 + (aMid - a1) * rho;
176  rWtd = r1 + (rMid - r1) * rho;
177 
178  // --- determine how much inertial damping to apply
179  if ( InertDamping == NO_DAMPING ) sigma = 1.0;
180  else if ( InertDamping == FULL_DAMPING ) sigma = 0.0;
181 
182  // --- use full inertial damping if closed conduit is surcharged
183  if ( isFull && !xsect_isOpen(xsect->type) ) sigma = 0.0;
184 
185  // --- compute terms of momentum eqn.:
186  // --- 1. friction slope term
187  if ( xsect->type == FORCE_MAIN && isFull )
188  dq1 = dt * forcemain_getFricSlope(j, fabs(v), rMid);
189  else dq1 = dt * Conduit[k].roughFactor / pow(rWtd, 1.33333) * fabs(v);
190 
191  // --- 2. energy slope term
192  dq2 = dt * GRAVITY * aWtd * (h2 - h1) / length;
193 
194  // --- 3 & 4. inertial terms
195  dq3 = 0.0;
196  dq4 = 0.0;
197  if ( sigma > 0.0 )
198  {
199  dq3 = 2.0 * v * (aMid - aOld) * sigma;
200  dq4 = dt * v * v * (a2 - a1) / length * sigma;
201  }
202 
203  // --- 5. local losses term
204  dq5 = 0.0;
205  if ( Conduit[k].hasLosses )
206  {
207  dq5 = findLocalLosses(j, a1, a2, aMid, qLast) / 2.0 / length * dt;
208  }
209 
210  // --- 6. term for evap and seepage losses per unit length
211  dq6 = link_getLossRate(j, qOld, dt) * 1.5 * dt * v / link_getLength(j); //(5.1.008)
212 
213  // --- combine terms to find new conduit flow
214  denom = 1.0 + dq1 + dq5;
215  q = (qOld - dq2 + dq3 + dq4 - dq6) / denom;
216 
217  // --- compute derivative of flow w.r.t. head
218  Link[j].dqdh = 1.0 / denom * GRAVITY * dt * aWtd / length * barrels;
219 
220  // --- check if any flow limitation applies
221  Link[j].inletControl = FALSE;
222  Link[j].normalFlow = FALSE;
223  if ( q > 0.0 )
224  {
225  // --- check for inlet controlled culvert flow
226  if ( xsect->culvertCode > 0 && !isFull )
227  q = culvert_getInflow(j, q, h1);
228 
229  // --- check for normal flow limitation based on surface slope & Fr
230  else
231  if ( y1 < Link[j].xsect.yFull &&
232  ( Link[j].flowClass == SUBCRITICAL ||
233  Link[j].flowClass == SUPCRITICAL )
234  ) q = checkNormalFlow(j, q, y1, y2, a1, r1);
235  }
236 
237  // --- apply under-relaxation weighting between new & old flows;
238  // --- do not allow change in flow direction without first being zero
239  if ( steps > 0 )
240  {
241  q = (1.0 - omega) * qLast + omega * q;
242  if ( q * qLast < 0.0 ) q = 0.001 * SGN(q);
243  }
244 
245  // --- check if user-supplied flow limit applies
246  if ( Link[j].qLimit > 0.0 )
247  {
248  if ( fabs(q) > Link[j].qLimit ) q = SGN(q) * Link[j].qLimit;
249  }
250 
251  // --- check for reverse flow with closed flap gate
252  if ( link_setFlapGate(j, n1, n2, q) ) q = 0.0;
253 
254  // --- do not allow flow out of a dry node
255  // (as suggested by R. Dickinson)
256  if( q > FUDGE && Node[n1].newDepth <= FUDGE ) q = FUDGE;
257  if( q < -FUDGE && Node[n2].newDepth <= FUDGE ) q = -FUDGE;
258 
259  // --- save new values of area, flow, depth, & volume
260  Conduit[k].a1 = aMid;
261  Conduit[k].q1 = q;
262  Conduit[k].q2 = q;
263  Link[j].newDepth = MIN(yMid, xsect->yFull);
264  aMid = (a1 + a2) / 2.0;
265  aMid = MIN(aMid, xsect->aFull);
266  Conduit[k].fullState = link_getFullState(a1, a2, xsect->aFull); //(5.1.008)
267  Link[j].newVolume = aMid * link_getLength(j) * barrels;
268  Link[j].newFlow = q * barrels;
269 }

273 int getFlowClass(int j, double q, double h1, double h2, double y1, double y2,
274  double *yC, double *yN, double* fasnh)
275 //
276 // Input: j = conduit link index
277 // q = current conduit flow (cfs)
278 // h1 = head at upstream end of conduit (ft)
279 // h2 = head at downstream end of conduit (ft)
280 // y1 = upstream flow depth in conduit (ft)
281 // y2 = downstream flow depth in conduit (ft)
282 // yC = critical flow depth (ft)
283 // yN = normal flow depth (ft)
284 // fasnh = fraction between norm. & crit. depth
285 // Output: returns flow classification code
286 // Purpose: determines flow class for a conduit based on depths at each end.
287 //
288 {
289  int n1, n2; // indexes of upstrm/downstrm nodes
290  int flowClass; // flow classification code
291  double ycMin, ycMax; // min/max critical depths (ft)
292  double z1, z2; // offsets of conduit inverts (ft)
293 
294  // --- get upstream & downstream node indexes
295  n1 = Link[j].node1;
296  n2 = Link[j].node2;
297 
298  // --- get upstream & downstream conduit invert offsets
299  z1 = Link[j].offset1;
300  z2 = Link[j].offset2;
301 
302  // --- base offset of an outfall conduit on outfall's depth
303  if ( Node[n1].type == OUTFALL ) z1 = MAX(0.0, (z1 - Node[n1].newDepth));
304  if ( Node[n2].type == OUTFALL ) z2 = MAX(0.0, (z2 - Node[n2].newDepth));
305 
306  // --- default class is SUBCRITICAL
307  flowClass = SUBCRITICAL;
308  *fasnh = 1.0;
309 
310  // --- case where both ends of conduit are wet
311  if ( y1 > FUDGE && y2 > FUDGE )
312  {
313  if ( q < 0.0 )
314  {
315  // --- upstream end at critical depth if flow depth is
316  // below conduit's critical depth and an upstream
317  // conduit offset exists
318  if ( z1 > 0.0 )
319  {
320  *yN = link_getYnorm(j, fabs(q));
321  *yC = link_getYcrit(j, fabs(q));
322  ycMin = MIN(*yN, *yC);
323  if ( y1 < ycMin ) flowClass = UP_CRITICAL;
324  }
325  }
326 
327  // --- case of normal direction flow
328  else
329  {
330  // --- downstream end at smaller of critical and normal depth
331  // if downstream flow depth below this and a downstream
332  // conduit offset exists
333  if ( z2 > 0.0 )
334  {
335  *yN = link_getYnorm(j, fabs(q));
336  *yC = link_getYcrit(j, fabs(q));
337  ycMin = MIN(*yN, *yC);
338  ycMax = MAX(*yN, *yC);
339  if ( y2 < ycMin ) flowClass = DN_CRITICAL;
340  else if ( y2 < ycMax )
341  {
342  if ( ycMax - ycMin < FUDGE ) *fasnh = 0.0;
343  else *fasnh = (ycMax - y2) / (ycMax - ycMin);
344  }
345  }
346  }
347  }
348 
349  // --- case where no flow at either end of conduit
350  else if ( y1 <= FUDGE && y2 <= FUDGE ) flowClass = DRY;
351 
352  // --- case where downstream end of pipe is wet, upstream dry
353  else if ( y2 > FUDGE )
354  {
355  // --- flow classification is UP_DRY if downstream head <
356  // invert of upstream end of conduit
357  if ( h2 < Node[n1].invertElev + Link[j].offset1 ) flowClass = UP_DRY;
358 
359  // --- otherwise, the downstream head will be >= upstream
360  // conduit invert creating a flow reversal and upstream end
361  // should be at critical depth, providing that an upstream
362  // offset exists (otherwise subcritical condition is maintained)
363  else if ( z1 > 0.0 )
364  {
365  *yN = link_getYnorm(j, fabs(q));
366  *yC = link_getYcrit(j, fabs(q));
367  flowClass = UP_CRITICAL;
368  }
369  }
370 
371  // --- case where upstream end of pipe is wet, downstream dry
372  else
373  {
374  // --- flow classification is DN_DRY if upstream head <
375  // invert of downstream end of conduit
376  if ( h1 < Node[n2].invertElev + Link[j].offset2 ) flowClass = DN_DRY;
377 
378  // --- otherwise flow at downstream end should be at critical depth
379  // providing that a downstream offset exists (otherwise
380  // subcritical condition is maintained)
381  else if ( z2 > 0.0 )
382  {
383  *yN = link_getYnorm(j, fabs(q));
384  *yC = link_getYcrit(j, fabs(q));
385  flowClass = DN_CRITICAL;
386  }
387  }
388  return flowClass;
389 }

393 void findSurfArea(int j, double q, double length, double* h1, double* h2,
394  double* y1, double* y2)
395 //
396 // Input: j = conduit link index
397 // q = current conduit flow (cfs)
398 // length = conduit length (ft)
399 // h1 = head at upstream end of conduit (ft)
400 // h2 = head at downstream end of conduit (ft)
401 // y1 = upstream flow depth (ft)
402 // y2 = downstream flow depth (ft)
403 // Output: updated values of h1, h2, y1, & y2;
404 // Purpose: assigns surface area of conduit to its up and downstream nodes.
405 //
406 {
407  int n1, n2; // indexes of upstrm/downstrm nodes
408  double flowDepth1; // flow depth at upstrm end (ft)
409  double flowDepth2; // flow depth at downstrm end (ft)
410  double flowDepthMid; // flow depth at midpt. (ft)
411  double width1; // top width at upstrm end (ft)
412  double width2; // top width at downstrm end (ft)
413  double widthMid; // top width at midpt. (ft)
414  double surfArea1 = 0.0; // surface area at upstream node (ft2)
415  double surfArea2 = 0.0; // surface area st downstrm node (ft2)
416  double criticalDepth; // critical flow depth (ft)
417  double normalDepth; // normal flow depth (ft)
418  double fasnh; // fraction between norm. & crit. depth
419  TXsect* xsect = &Link[j].xsect; // pointer to cross-section data
420 
421  // --- get node indexes & current flow depths
422  n1 = Link[j].node1;
423  n2 = Link[j].node2;
424  flowDepth1 = *y1;
425  flowDepth2 = *y2;
426 
427  normalDepth = (flowDepth1 + flowDepth2) / 2.0;
428  criticalDepth = normalDepth;
429 
430  // --- find conduit's flow classification
431  Link[j].flowClass = getFlowClass(j, q, *h1, *h2, *y1, *y2,
432  &criticalDepth, &normalDepth, &fasnh);
433 
434  // --- add conduit's surface area to its end nodes depending on flow class
435  switch ( Link[j].flowClass )
436  {
437  case SUBCRITICAL:
438  flowDepthMid = 0.5 * (flowDepth1 + flowDepth2);
439  if ( flowDepthMid < FUDGE ) flowDepthMid = FUDGE;
440  width1 = getWidth(xsect, flowDepth1);
441  width2 = getWidth(xsect, flowDepth2);
442  widthMid = getWidth(xsect, flowDepthMid);
443  surfArea1 = (width1 + widthMid) * length / 4.;
444  surfArea2 = (widthMid + width2) * length / 4. * fasnh;
445  break;
446 
447  case UP_CRITICAL:
448  flowDepth1 = criticalDepth;
449  if ( normalDepth < criticalDepth ) flowDepth1 = normalDepth;
450  flowDepth1 = MAX(flowDepth1, FUDGE);
451  *h1 = Node[n1].invertElev + Link[j].offset1 + flowDepth1;
452  flowDepthMid = 0.5 * (flowDepth1 + flowDepth2);
453  if ( flowDepthMid < FUDGE ) flowDepthMid = FUDGE;
454  width2 = getWidth(xsect, flowDepth2);
455  widthMid = getWidth(xsect, flowDepthMid);
456  surfArea2 = (widthMid + width2) * length * 0.5;
457  break;
458 
459  case DN_CRITICAL:
460  flowDepth2 = criticalDepth;
461  if ( normalDepth < criticalDepth ) flowDepth2 = normalDepth;
462  flowDepth2 = MAX(flowDepth2, FUDGE);
463  *h2 = Node[n2].invertElev + Link[j].offset2 + flowDepth2;
464  width1 = getWidth(xsect, flowDepth1);
465  flowDepthMid = 0.5 * (flowDepth1 + flowDepth2);
466  if ( flowDepthMid < FUDGE ) flowDepthMid = FUDGE;
467  widthMid = getWidth(xsect, flowDepthMid);
468  surfArea1 = (width1 + widthMid) * length * 0.5;
469  break;
470 
471  case UP_DRY:
472  flowDepth1 = FUDGE;
473  flowDepthMid = 0.5 * (flowDepth1 + flowDepth2);
474  if ( flowDepthMid < FUDGE ) flowDepthMid = FUDGE;
475  width1 = getWidth(xsect, flowDepth1);
476  width2 = getWidth(xsect, flowDepth2);
477  widthMid = getWidth(xsect, flowDepthMid);
478 
479  // --- assign avg. surface area of downstream half of conduit
480  // to the downstream node
481  surfArea2 = (widthMid + width2) * length / 4.;
482 
483  // --- if there is no free-fall at upstream end, assign the
484  // upstream node the avg. surface area of the upstream half
485  if ( Link[j].offset1 <= 0.0 )
486  {
487  surfArea1 = (width1 + widthMid) * length / 4.;
488  }
489  break;
490 
491  case DN_DRY:
492  flowDepth2 = FUDGE;
493  flowDepthMid = 0.5 * (flowDepth1 + flowDepth2);
494  if ( flowDepthMid < FUDGE ) flowDepthMid = FUDGE;
495  width1 = getWidth(xsect, flowDepth1);
496  width2 = getWidth(xsect, flowDepth2);
497  widthMid = getWidth(xsect, flowDepthMid);
498 
499  // --- assign avg. surface area of upstream half of conduit
500  // to the upstream node
501  surfArea1 = (widthMid + width1) * length / 4.;
502 
503  // --- if there is no free-fall at downstream end, assign the
504  // downstream node the avg. surface area of the downstream half
505  if ( Link[j].offset2 <= 0.0 )
506  {
507  surfArea2 = (width2 + widthMid) * length / 4.;
508  }
509  break;
510 
511  case DRY:
512  surfArea1 = FUDGE * length / 2.0;
513  surfArea2 = surfArea1;
514  break;
515  }
516  Link[j].surfArea1 = surfArea1;
517  Link[j].surfArea2 = surfArea2;
518  *y1 = flowDepth1;
519  *y2 = flowDepth2;
520 }

524 double findLocalLosses(int j, double a1, double a2, double aMid, double q)
525 //
526 // Input: j = link index
527 // a1 = upstream area (ft2)
528 // a2 = downstream area (ft2)
529 // aMid = midpoint area (ft2)
530 // q = flow rate (cfs)
531 // Output: returns local losses (ft/sec)
532 // Purpose: computes local losses term of momentum equation.
533 //
534 {
535  double losses = 0.0;
536  q = fabs(q);
537  if ( a1 > FUDGE ) losses += Link[j].cLossInlet * (q/a1);
538  if ( a2 > FUDGE ) losses += Link[j].cLossOutlet * (q/a2);
539  if ( aMid > FUDGE ) losses += Link[j].cLossAvg * (q/aMid);
540  return losses;
541 }

577 double getHydRad(TXsect* xsect, double y)
578 //
579 // Input: xsect = ptr. to conduit cross section
580 // y = flow depth (ft)
581 // Output: returns hydraulic radius (ft)
582 // Purpose: computes hydraulic radius of flow cross-section in a conduit.
583 //
584 {
