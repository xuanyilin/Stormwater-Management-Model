53 int kinwave_execute(int j, double* qinflow, double* qoutflow, double tStep)
54 //
55 // Input: j = link index
56 // qinflow = inflow at current time (cfs)
57 // tStep = time step (sec)
58 // Output: qoutflow = outflow at current time (cfs),
59 // returns number of iterations used
60 // Purpose: finds outflow over time step tStep given flow entering a
61 // conduit using Kinematic Wave flow routing.
62 //
63 //
64 // ^ q3
65 // t |
66 // | qin, ain |-------------------| qout, aout
67 // | | Flow ---> |
68 // |----> x q1, a1 |-------------------| q2, a2
69 //
70 //
71 {
72  int k;
73  int result = 1;
74  double dxdt, dq;
75  double ain, aout;
76  double qin, qout;
77  double a1, a2, q1, q2, q3;
78 
79  // --- no routing for non-conduit link
80  (*qoutflow) = (*qinflow);
81  if ( Link[j].type != CONDUIT ) return result;
82 
83  // --- no routing for dummy xsection
84  if ( Link[j].xsect.type == DUMMY ) return result;
85 
86  // --- assign module-level variables
87  pXsect = &Link[j].xsect;
88  Qfull = Link[j].qFull;
89  Afull = Link[j].xsect.aFull;
90  k = Link[j].subIndex;
91  Beta1 = Conduit[k].beta / Qfull;
92 
93  // --- normalize previous flows
94  q1 = Conduit[k].q1 / Qfull;
95  q2 = Conduit[k].q2 / Qfull;
96 
97  // --- normalize inflow //(5.1.008)
98  qin = (*qinflow) / Conduit[k].barrels / Qfull;
99 
100  // --- compute evaporation and infiltration loss rate
101  q3 = link_getLossRate(j, qin*Qfull, tStep) / Qfull; //(5.1.008)
102 
103  // --- normalize previous areas
104  a1 = Conduit[k].a1 / Afull;
105  a2 = Conduit[k].a2 / Afull;
106 
107  // --- use full area when inlet flow >= full flow
108  if ( qin >= 1.0 ) ain = 1.0;
109 
110  // --- get normalized inlet area corresponding to inlet flow
111  else ain = xsect_getAofS(pXsect, qin/Beta1) / Afull;
112 
113  // --- check for no flow
114  if ( qin <= TINY && q2 <= TINY )
115  {
116  qout = 0.0;
117  aout = 0.0;
118  }
119 
120  // --- otherwise solve finite difference form of continuity eqn.
121  else
122  {
123  // --- compute constant factors
124  dxdt = link_getLength(j) / tStep * Afull / Qfull;
125  dq = q2 - q1;
126  C1 = dxdt * WT / WX;
127  C2 = (1.0 - WT) * (ain - a1);
128  C2 = C2 - WT * a2;
129  C2 = C2 * dxdt / WX;
130  C2 = C2 + (1.0 - WX) / WX * dq - qin;
131  C2 = C2 + q3 / WX;
132 
133  // --- starting guess for aout is value from previous time step
134  aout = a2;
135 
136  // --- solve continuity equation for aout
137  result = solveContinuity(qin, ain, &aout);
138 
139  // --- report error if continuity eqn. not solved
140  if ( result == -1 )
141  {
142  report_writeErrorMsg(ERR_KINWAVE, Link[j].ID);
143  return 1;
144  }
145  if ( result <= 0 ) result = 1;
146 
147  // --- compute normalized outlet flow from outlet area
148  qout = Beta1 * xsect_getSofA(pXsect, aout*Afull);
149  if ( qin > 1.0 ) qin = 1.0;
150  }
151 
152  // --- save new flows and areas
153  Conduit[k].q1 = qin * Qfull;
154  Conduit[k].a1 = ain * Afull;
155  Conduit[k].q2 = qout * Qfull;
156  Conduit[k].a2 = aout * Afull;
157  Conduit[k].fullState =
158  link_getFullState(Conduit[k].a1, Conduit[k].a2, Afull); //(5.1.008)
159  (*qinflow) = Conduit[k].q1 * Conduit[k].barrels;
160  (*qoutflow) = Conduit[k].q2 * Conduit[k].barrels;
161  return result;
162 }

166 int solveContinuity(double qin, double ain, double* aout)
167 //
168 // Input: qin = upstream normalized flow
169 // ain = upstream normalized area
170 // aout = downstream normalized area
171 // Output: new value for aout; returns an error code
172 // Purpose: solves continuity equation f(a) = Beta1*S(a) + C1*a + C2 = 0
173 // for 'a' using the Newton-Raphson root finder function.
174 // Return code has the following meanings:
175 // >= 0 number of function evaluations used
176 // -1 Newton function failed
177 // -2 flow always above max. flow
178 // -3 flow always below zero
179 //
180 // Note: pXsect (pointer to conduit's cross-section), and constants Beta1,
181 // C1, and C2 are module-level shared variables assigned values
182 // in kinwave_execute().
183 //
184 {
185  int n; // # evaluations or error code
186  double aLo, aHi, aTmp; // lower/upper bounds on a
187  double fLo, fHi; // lower/upper bounds on f
188  double tol = EPSIL; // absolute convergence tol.
189 
190  // --- first determine bounds on 'a' so that f(a) passes through 0.
191 
192  // --- set upper bound to area at full flow
193  aHi = 1.0;
194  fHi = 1.0 + C1 + C2;
195 
196  // --- try setting lower bound to area where section factor is maximum
197  aLo = xsect_getAmax(pXsect) / Afull;
198  if ( aLo < aHi )
199  {
200  fLo = ( Beta1 * pXsect->sMax ) + (C1 * aLo) + C2;
201  }
202  else fLo = fHi;
203 
204  // --- if fLo and fHi have same sign then set lower bound to 0
205  if ( fHi*fLo > 0.0 )
206  {
207  aHi = aLo;
208  fHi = fLo;
209  aLo = 0.0;
210  fLo = C2;
211  }
212 
213  // --- proceed with search for root if fLo and fHi have different signs
214  if ( fHi*fLo <= 0.0 )
215  {
216  // --- start search at midpoint of lower/upper bounds
217  // if initial value outside of these bounds
218  if ( *aout < aLo || *aout > aHi ) *aout = 0.5*(aLo + aHi);
219 
220  // --- if fLo > fHi then switch aLo and aHi
221  if ( fLo > fHi )
222  {
223  aTmp = aLo;
224  aLo = aHi;
225  aHi = aTmp;
226  }
227 
228  // --- call the Newton root finder method passing it the
229  // evalContinuity function to evaluate the function
230  // and its derivatives
231  n = findroot_Newton(aLo, aHi, aout, tol, evalContinuity, NULL);
232 
233  // --- check if root finder succeeded
234  if ( n <= 0 ) n = -1;
235  }
236 
237  // --- if lower/upper bound functions both negative then use full flow
238  else if ( fLo < 0.0 )
239  {
240  if ( qin > 1.0 ) *aout = ain;
241  else *aout = 1.0;
242  n = -2;
243  }
244 
245  // --- if lower/upper bound functions both positive then use no flow
246  else if ( fLo > 0 )
247  {
248  *aout = 0.0;
249  n = -3;
250  }
251  else n = -1;
252  return n;
253 }

257 void evalContinuity(double a, double* f, double* df, void* p)
258 //
259 // Input: a = outlet normalized area
260 // Output: f = value of continuity eqn.
261 // df = derivative of continuity eqn.
262 // Purpose: computes value of continuity equation (f) and its derivative (df)
263 // w.r.t. normalized area for link with normalized outlet area 'a'.
264 //
265 {
266  *f = (Beta1 * xsect_getSofA(pXsect, a*Afull)) + (C1 * a) + C2;
267  *df = (Beta1 * Afull * xsect_getdSdA(pXsect, a*Afull)) + C1;
268 }
