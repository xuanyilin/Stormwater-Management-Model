51 int DLLEXPORT swmm_getSimulationDateTime(int timetype, int *year, int *month, int *day,
52  int *hours, int *minutes, int *seconds)
53 //
54 // Input: timetype = time type to return
55 // Output: year, month, day, hours, minutes, seconds = int
56 // Return: API Error
57 // Purpose: Get the simulation start, end and report date times
58 {
59  int errcode = 0;
60  // Check if Open
61  if (swmm_IsOpenFlag() == FALSE)
62  {
63  errcode = ERR_API_INPUTNOTOPEN;
64  }
65  else
66  {
67  DateTime _dtime;
68  switch (timetype)
69  {
70  //StartDateTime (globals.h)
71  case SM_STARTDATE: _dtime = StartDateTime; break;
72  //EndDateTime (globals.h)
73  case SM_ENDDATE: _dtime = EndDateTime; break;
74  //ReportStart (globals.h)
75  case SM_REPORTDATE: _dtime = ReportStart; break;
76  default: return(ERR_API_OUTBOUNDS);
77  }
78  datetime_decodeDate(_dtime, year, month, day);
79  datetime_decodeTime(_dtime, hours, minutes, seconds);
80  }
81 
82  return (errcode);
83 }

85 int DLLEXPORT swmm_setSimulationDateTime(int timetype, char *dtimestr)
86 //
87 // Input: timetype = time type to return
88 // DateTime String
89 // Return: API Error
90 // Purpose: Get the simulation start, end and report date times
91 {
92  int errcode = 0;
93 
94  char theDate[10];
95  char theTime[9];
96 
97  // Check if Open
98  if(swmm_IsOpenFlag() == FALSE)
99  {
100  errcode = ERR_API_INPUTNOTOPEN;
101  }
102  // Check if Simulation is Running
103  else if(swmm_IsStartedFlag() == TRUE)
104  {
105  errcode = ERR_API_SIM_NRUNNING;
106  }
107  else
108  {
109  strncpy(theDate, dtimestr, 10);
110  strncpy(theTime, dtimestr+11, 9);
111 
112  switch(timetype)
113  {
114  //StartDateTime (globals.h)
115  case SM_STARTDATE:
116  project_readOption("START_DATE", theDate);
117  project_readOption("START_TIME", theTime);
118  StartDateTime = StartDate + StartTime;
119  TotalDuration = floor((EndDateTime - StartDateTime) * SECperDAY);
120  // --- convert total duration to milliseconds
121  TotalDuration *= 1000.0;
122  break;
123  //EndDateTime (globals.h)
124  case SM_ENDDATE:
125  project_readOption("END_DATE", theDate);
126  project_readOption("END_TIME", theTime);
127  EndDateTime = EndDate + EndTime;
128  TotalDuration = floor((EndDateTime - StartDateTime) * SECperDAY);
129  // --- convert total duration to milliseconds
130  TotalDuration *= 1000.0;
131  break;
132  //ReportStart (globals.h)
133  case SM_REPORTDATE:
134  project_readOption("REPORT_START_DATE", theDate);
135  project_readOption("REPORT_START_TIME", theTime);
136  ReportStart = ReportStartDate + ReportStartTime;
137  break;
138  default: errcode = ERR_API_OUTBOUNDS; break;
139  }
140  }
141 
142  return (errcode);
143 }
