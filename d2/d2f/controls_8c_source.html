13 // Rule-based controls functions.
14 //
15 // Control rules have the format:
16 // RULE name
17 // IF <premise>
18 // AND / OR <premise>
19 // etc.
20 // THEN <action>
21 // AND <action>
22 // etc.
23 // ELSE <action>
24 // AND <action>
25 // etc.
26 // PRIORITY <p>
27 //
28 // <premise> consists of:
29 // <variable> <relational operator> value / <variable>
30 // where <variable> is <object type> <id name> <attribute>
31 // E.g.: Node 123 Depth > 4.5
32 // Node 456 Depth < Node 123 Depth
33 //
34 // <action> consists of:
35 // <variable> = setting
36 // E.g.: Pump abc status = OFF
37 // Weir xyz setting = 0.5
38 //
39 // Build 5.1.008:
40 // - Support added for r.h.s. variables in rule premises.
41 // - Node volume added as a premise variable.
42 //
43 // Build 5.1.009:
44 // - Fixed problem with parsing a RHS premise variable.
45 //
46 // Build 5.1.010:
47 // - Support added for link TIMEOPEN & TIMECLOSED premises.
48 //

191 int controls_create(int n)
192 //
193 // Input: n = total number of control rules
194 // Output: returns error code
195 // Purpose: creates an array of control rules.
196 //
197 {
198  int r;
199  ActionList = NULL;
200  InputState = r_PRIORITY;
201  RuleCount = n;
202  if ( n == 0 ) return 0;
203  Rules = (struct TRule *) calloc(RuleCount, sizeof(struct TRule));
204  if (Rules == NULL) return ERR_MEMORY;
205  for ( r=0; r<RuleCount; r++ )
206  {
207  Rules[r].ID = NULL;
208  Rules[r].firstPremise = NULL;
209  Rules[r].lastPremise = NULL;
210  Rules[r].thenActions = NULL;
211  Rules[r].elseActions = NULL;
212  Rules[r].priority = 0.0;
213  }
214  return 0;
215 }

233 int controls_addRuleClause(int r, int keyword, char* tok[], int nToks)
234 //
235 // Input: r = rule index
236 // keyword = the clause's keyword code (IF, THEN, etc.)
237 // tok = an array of string tokens that comprises the clause
238 // nToks = number of tokens
239 // Output: returns an error code
240 // Purpose: addd a new clause to a control rule.
241 //
242 {
243  switch (keyword)
244  {
245  case r_RULE:
246  if ( Rules[r].ID == NULL )
247  Rules[r].ID = project_findID(CONTROL, tok[1]);
248  InputState = r_RULE;
249  if ( nToks > 2 ) return ERR_RULE;
250  return 0;
251 
252  case r_IF:
253  if ( InputState != r_RULE ) return ERR_RULE;
254  InputState = r_IF;
255  return addPremise(r, r_AND, tok, nToks);
256 
257  case r_AND:
258  if ( InputState == r_IF ) return addPremise(r, r_AND, tok, nToks);
259  else if ( InputState == r_THEN || InputState == r_ELSE )
260  return addAction(r, tok, nToks);
261  else return ERR_RULE;
262 
263  case r_OR:
264  if ( InputState != r_IF ) return ERR_RULE;
265  return addPremise(r, r_OR, tok, nToks);
266 
267  case r_THEN:
268  if ( InputState != r_IF ) return ERR_RULE;
269  InputState = r_THEN;
270  return addAction(r, tok, nToks);
271 
272  case r_ELSE:
273  if ( InputState != r_THEN ) return ERR_RULE;
274  InputState = r_ELSE;
275  return addAction(r, tok, nToks);
276 
277  case r_PRIORITY:
278  if ( InputState != r_THEN && InputState != r_ELSE ) return ERR_RULE;
279  InputState = r_PRIORITY;
280  if ( !getDouble(tok[1], &Rules[r].priority) ) return ERR_NUMBER;
281  if ( nToks > 2 ) return ERR_RULE;
282  return 0;
283  }
284  return 0;
285 }

289 int controls_evaluate(DateTime currentTime, DateTime elapsedTime, double tStep)
290 //
291 // Input: currentTime = current simulation date/time
292 // elapsedTime = decimal days since start of simulation
293 // tStep = simulation time step (days)
294 // Output: returns number of new actions taken
295 // Purpose: evaluates all control rules at current time of the simulation.
296 //
297 {
298  int r; // control rule index
299  int result; // TRUE if rule premises satisfied
300  struct TPremise* p; // pointer to rule premise clause
301  struct TAction* a; // pointer to rule action clause
302 
303  // --- save date and time to shared variables
304  CurrentDate = floor(currentTime);
305  CurrentTime = currentTime - floor(currentTime);
306  ElapsedTime = elapsedTime;
307 
308  // --- evaluate each rule
309  if ( RuleCount == 0 ) return 0;
310  clearActionList();
311  for (r=0; r<RuleCount; r++)
312  {
313  // --- evaluate rule's premises
314  result = TRUE;
315  p = Rules[r].firstPremise;
316  while (p)
317  {
318  if ( p->type == r_OR )
319  {
320  if ( result == FALSE )
321  result = evaluatePremise(p, tStep);
322  }
323  else
324  {
325  if ( result == FALSE ) break;
326  result = evaluatePremise(p, tStep);
327  }
328  p = p->next;
329  }
330 
331  // --- if premises true, add THEN clauses to action list
332  // else add ELSE clauses to action list
333  if ( result == TRUE ) a = Rules[r].thenActions;
334  else a = Rules[r].elseActions;
335  while (a)
336  {
337  updateActionValue(a, currentTime, tStep);
338  updateActionList(a);
339  a = a->next;
340  }
341  }
342 
343  // --- execute actions on action list
344  if ( ActionList ) return executeActionList(currentTime);
345  else return 0;
346 }

352 int addPremise(int r, int type, char* tok[], int nToks)
353 //
354 // Input: r = control rule index
355 // type = type of premise (IF, AND, OR)
356 // tok = array of string tokens containing premise statement
357 // nToks = number of string tokens
358 // Output: returns an error code
359 // Purpose: adds a new premise to a control rule.
360 //
361 {
362  int relation, n, err = 0;
363  double value = MISSING;
364  struct TPremise* p;
365  struct TVariable v1;
366  struct TVariable v2;
367 
368  // --- check for minimum number of tokens
369  if ( nToks < 5 ) return ERR_ITEMS;
370 
371  // --- get LHS variable
372  n = 1;
373  err = getPremiseVariable(tok, &n, &v1);
374  if ( err > 0 ) return err;
375 
376  // --- get relational operator
377  n++;
378  relation = findExactMatch(tok[n], RelOpWords);
379  if ( relation < 0 ) return error_setInpError(ERR_KEYWORD, tok[n]);
380  n++;
381 
382  // --- initialize RHS variable
383  v2.attribute = -1;
384  v2.link = -1;
385  v2.node = -1;
386 
387  // --- check that more tokens remain
388  if ( n >= nToks ) return error_setInpError(ERR_ITEMS, "");
389 
390  // --- see if a RHS variable is supplied
391  if ( findmatch(tok[n], ObjectWords) >= 0 && n + 3 >= nToks )
392  {
393  err = getPremiseVariable(tok, &n, &v2);
394  if ( err > 0 ) return ERR_RULE; //(5.1.009)
395  if ( v1.attribute != v2.attribute) //(5.1.009)
396  report_writeWarningMsg(WARN11, Rules[r].ID); //(5.1.009)
397  }
398 
399  // --- otherwise get value to which LHS variable is compared to
400  else
401  {
402  err = getPremiseValue(tok[n], v1.attribute, &value);
403  n++;
404  }
405  if ( err > 0 ) return err;
406 
407  // --- make sure another clause is not on same line
408  if ( n < nToks && findmatch(tok[n], RuleKeyWords) >= 0 ) return ERR_RULE;
409 
410  // --- create the premise object
411  p = (struct TPremise *) malloc(sizeof(struct TPremise));
412  if ( !p ) return ERR_MEMORY;
413  p->type = type;
414  p->lhsVar = v1;
415  p->rhsVar = v2;
416  p->relation = relation;
417  p->value = value;
418  p->next = NULL;
419  if ( Rules[r].firstPremise == NULL )
420  {
421  Rules[r].firstPremise = p;
422  }
423  else
424  {
425  Rules[r].lastPremise->next = p;
426  }
427  Rules[r].lastPremise = p;
428  return 0;
429 }

433 int getPremiseVariable(char* tok[], int* k, struct TVariable* v)
434 //
435 // Input: tok = array of string tokens containing premise statement
436 // k = index of current token
437 // Output: returns an error code; updates k to new current token and
438 // places identity of specified variable in v
439 // Purpose: parses a variable (e.g., Node 123 Depth) specified in a
440 // premise clause of a control rule.
441 //
442 {
443  int n = *k;
444  int node = -1;
445  int link = -1;
446  int obj, attrib;
447 
448  // --- get object type
449  obj = findmatch(tok[n], ObjectWords);
450  if ( obj < 0 ) return error_setInpError(ERR_KEYWORD, tok[n]);
451 
452  // --- get object index from its name
453  n++;
454  switch (obj)
455  {
456  case r_NODE:
457  node = project_findObject(NODE, tok[n]);
458  if ( node < 0 ) return error_setInpError(ERR_NAME, tok[n]);
459  break;
460 
461  case r_LINK:
462  case r_CONDUIT:
463  case r_PUMP:
464  case r_ORIFICE:
465  case r_WEIR:
466  case r_OUTLET:
467  link = project_findObject(LINK, tok[n]);
468  if ( link < 0 ) return error_setInpError(ERR_NAME, tok[n]);
469  break;
470  default: n--;
471  }
472  n++;
473 
474  // --- get attribute index from its name
475  attrib = findmatch(tok[n], AttribWords);
476  if ( attrib < 0 ) return error_setInpError(ERR_KEYWORD, tok[n]);
477 
478  // --- check that attribute belongs to object type
479  if ( obj == r_NODE ) switch (attrib)
480  {
481  case r_DEPTH:
482  case r_HEAD:
483  case r_VOLUME: //(5.1.008)
484  case r_INFLOW: break;
485  default: return error_setInpError(ERR_KEYWORD, tok[n]);
486  }
487 
489  // --- check for link TIMEOPEN & TIMECLOSED attributes
490  else if ( link >= 0 &&
491  ( (attrib == r_TIMEOPEN ||
492  attrib == r_TIMECLOSED)
493  ))
494  {
495 
496  }
498 
499  else if ( obj == r_LINK || obj == r_CONDUIT ) switch (attrib)
500  {
501  case r_STATUS:
502  case r_DEPTH:
503  case r_FLOW: break;
504  default: return error_setInpError(ERR_KEYWORD, tok[n]);
505  }
506  else if ( obj == r_PUMP ) switch (attrib)
507  {
508  case r_FLOW:
509  case r_STATUS: break;
510  default: return error_setInpError(ERR_KEYWORD, tok[n]);
511  }
512  else if ( obj == r_ORIFICE || obj == r_WEIR ||
513  obj == r_OUTLET ) switch (attrib)
514  {
515  case r_SETTING: break;
516  default: return error_setInpError(ERR_KEYWORD, tok[n]);
517  }
518  else switch (attrib)
519  {
520  case r_TIME:
521  case r_DATE:
522  case r_CLOCKTIME:
523  case r_DAY:
524  case r_MONTH:
525  case r_DAYOFYEAR: break; //(5.1.011)
526  default: return error_setInpError(ERR_KEYWORD, tok[n]);
527  }
528 
529  // --- populate variable structure
530  v->node = node;
531  v->link = link;
532  v->attribute = attrib;
533  *k = n;
534  return 0;
535 }

539 int getPremiseValue(char* token, int attrib, double* value)
540 //
541 // Input: token = a string token
542 // attrib = index of a node/link attribute
543 // Output: value = attribute value;
544 // returns an error code;
545 // Purpose: parses the numerical value of a particular node/link attribute
546 // in the premise clause of a control rule.
547 //
548 {
549  char strDate[25]; //(5.1.011)
550  switch (attrib)
551  {
552  case r_STATUS:
553  *value = findmatch(token, StatusWords);
554  if ( *value < 0.0 ) *value = findmatch(token, ConduitWords);
555  if ( *value < 0.0 ) return error_setInpError(ERR_KEYWORD, token);
556  break;
557 
558  case r_TIME:
559  case r_CLOCKTIME:
560  case r_TIMEOPEN: //(5.1.010)
561  case r_TIMECLOSED: //(5.1.010)
562  if ( !datetime_strToTime(token, value) )
563  return error_setInpError(ERR_DATETIME, token);
564  break;
565 
566  case r_DATE:
567  if ( !datetime_strToDate(token, value) )
568  return error_setInpError(ERR_DATETIME, token);
569  break;
570 
571  case r_DAY:
572  if ( !getDouble(token, value) )
573  return error_setInpError(ERR_NUMBER, token);
574  if ( *value < 1.0 || *value > 7.0 )
575  return error_setInpError(ERR_DATETIME, token);
576  break;
577 
578  case r_MONTH:
579  if ( !getDouble(token, value) )
580  return error_setInpError(ERR_NUMBER, token);
581  if ( *value < 1.0 || *value > 12.0 )
582  return error_setInpError(ERR_DATETIME, token);
583  break;
584 
586  case r_DAYOFYEAR:
587  strncpy(strDate, token, 6);
588  strcat(strDate, "/1947");
589  if ( datetime_strToDate(strDate, value) )
590  {
591  *value = datetime_dayOfYear(*value);
592  }
593  else if ( !getDouble(token, value) || *value < 1 || *value > 365 )
594  return error_setInpError(ERR_DATETIME, token);
595  break;
597 
598  default: if ( !getDouble(token, value) )
599  return error_setInpError(ERR_NUMBER, token);
600  }
601  return 0;
602 }

606 int addAction(int r, char* tok[], int nToks)
607 //
608 // Input: r = control rule index
609 // tok = array of string tokens containing action statement
610 // nToks = number of string tokens
611 // Output: returns an error code
612 // Purpose: adds a new action to a control rule.
613 //
614 {
615  int obj, link, attrib;
616  int curve = -1, tseries = -1;
617  int n;
618  int err;
619  double values[] = {1.0, 0.0, 0.0};
620 
621  struct TAction* a;
622 
623  // --- check for proper number of tokens
624  if ( nToks < 6 ) return error_setInpError(ERR_ITEMS, "");
625 
626  // --- check for valid object type
627  obj = findmatch(tok[1], ObjectWords);
628  if ( obj != r_LINK && obj != r_CONDUIT && obj != r_PUMP &&
629  obj != r_ORIFICE && obj != r_WEIR && obj != r_OUTLET )
630  return error_setInpError(ERR_KEYWORD, tok[1]);
631 
632  // --- check that object name exists and is of correct type
633  link = project_findObject(LINK, tok[2]);
634  if ( link < 0 ) return error_setInpError(ERR_NAME, tok[2]);
635  switch (obj)
636  {
637  case r_CONDUIT:
638  if ( Link[link].type != CONDUIT )
639  return error_setInpError(ERR_NAME, tok[2]);
640  break;
641  case r_PUMP:
642  if ( Link[link].type != PUMP )
643  return error_setInpError(ERR_NAME, tok[2]);
644  break;
645  case r_ORIFICE:
646  if ( Link[link].type != ORIFICE )
647  return error_setInpError(ERR_NAME, tok[2]);
648  break;
649  case r_WEIR:
650  if ( Link[link].type != WEIR )
651  return error_setInpError(ERR_NAME, tok[2]);
652  break;
653  case r_OUTLET:
654  if ( Link[link].type != OUTLET )
655  return error_setInpError(ERR_NAME, tok[2]);
656  break;
657  }
658 
659  // --- check for valid attribute name
660  attrib = findmatch(tok[3], AttribWords);
661  if ( attrib < 0 ) return error_setInpError(ERR_KEYWORD, tok[3]);
662 
663  // --- get control action setting
664  if ( obj == r_CONDUIT )
665  {
666  if ( attrib == r_STATUS )
667  {
668  values[0] = findmatch(tok[5], ConduitWords);
669  if ( values[0] < 0.0 )
670  return error_setInpError(ERR_KEYWORD, tok[5]);
671  }
672  else return error_setInpError(ERR_KEYWORD, tok[3]);
673  }
674 
675  else if ( obj == r_PUMP )
676  {
677  if ( attrib == r_STATUS )
678  {
679  values[0] = findmatch(tok[5], StatusWords);
680  if ( values[0] < 0.0 )
681  return error_setInpError(ERR_KEYWORD, tok[5]);
682  }
683  else if ( attrib == r_SETTING )
684  {
685  err = setActionSetting(tok, nToks, &curve, &tseries,
686  &attrib, values);
687  if ( err > 0 ) return err;
688  }
689  else return error_setInpError(ERR_KEYWORD, tok[3]);
690  }
691 
692  else if ( obj == r_ORIFICE || obj == r_WEIR || obj == r_OUTLET )
693  {
694  if ( attrib == r_SETTING )
695  {

828 double getPIDSetting(struct TAction* a, double dt)
829 //
830 // Input: a = an action object
831 // dt = current time step (days)
832 // Output: returns a new link setting
833 // Purpose: computes a new setting for a link subject to a PID controller.
834 //
835 // Note: a->kp = gain coefficient,
836 // a->ki = integral time (minutes)
837 // a->k2 = derivative time (minutes)
838 // a->e1 = error from previous time step
839 // a->e2 = error from two time steps ago
840 {
841  double e0, setting;
842  double p, i, d, update;
843  double tolerance = 0.0001;
844 
845  // --- convert time step from days to minutes
846  dt *= 1440.0;
847 
848  // --- determine relative error in achieving controller set point
849  e0 = SetPoint - ControlValue;
850  if ( fabs(e0) > TINY )
851  {
852  if ( SetPoint != 0.0 ) e0 = e0/SetPoint;
853  else e0 = e0/ControlValue;
854  }
855 
856  // --- reset previous errors to 0 if controller gets stuck
857  if (fabs(e0 - a->e1) < tolerance)
858  {
859  a->e2 = 0.0;
860  a->e1 = 0.0;
861  }
862 
863  // --- use the recursive form of the PID controller equation to
864  // determine the new setting for the controlled link
865  p = (e0 - a->e1);
866  if ( a->ki == 0.0 ) i = 0.0;
867  else i = e0 * dt / a->ki;
868  d = a->kd * (e0 - 2.0*a->e1 + a->e2) / dt;
869  update = a->kp * (p + i + d);
870  if ( fabs(update) < tolerance ) update = 0.0;
871  setting = Link[a->link].targetSetting + update;
872 
873  // --- update previous errors
874  a->e2 = a->e1;
875  a->e1 = e0;
876 
877  // --- check that new setting lies within feasible limits
878  if ( setting < 0.0 ) setting = 0.0;
879  if (Link[a->link].type != PUMP && setting > 1.0 ) setting = 1.0;
880  return setting;
881 }
